245 /**
246  * Clase Bullet que construye un Bullet
247  * @class Bullet
248  * @constructors
249  * @this {Bullet}
250  * @param {Number} x Posicion X en el que se encuentra
251  * @param {Number} y Posicion Y en el que se encuentra
252  * @param {Enemy} Enemy enemigo que tiene seleccionado
253  * @param {Number} damage dano que inflije la bala
254  * @param {Number} index indice para sacar la informacion del json
255  * @param {String} direccion  direccion que va a disprar
256  * @returns {Bullet}
257  */
258 function Bullet(x, y, Enemy, damage, index, direccion) {
259 /* * Posicion X en el que se encuentra. Por defecto 0.
260    * @type Number
261    */ 
262     this.x = (x == null) ? 0 : x;
263     /* * Posicion Y en el que se encuentra. Por defecto 0.
264    * @type Number
265    */ 
266     this.y = (y == null) ? 0 : y;
267     /* * Posicion X inicial coje el valor de X al construir el objeto.
268    * @type Number
269    */ 
270     this.xi = this.x;
271      /* * Posicion Y inicial coje el valor de Y al construir el objeto.
272    * @type Number
273    */ 
274     this.yi = this.y;
275     /**
276      * Direccion donde va disparar "bot" || "top" || "left" || "right" solo se usara si bulletType es  "lineal" o "area".
277      * @type String
278      */
279     this.direccion = direccion;
280     /**
281      * Tipo de la bala coje el valor del json tipotower.
282      * @type String
283      */
284     this.bulletType = tipotower[index].bulletType;
285     /**
286      * Anchura de la bala coje el valor del json tipotower.
287      * @type Number
288      */
289     this.width = tipotower[index].bulletSize;
290     /**
291      * Altura de la bala coje el valor del json tipotower.
292      * @type Number
293      * 
294      */
295     this.height = tipotower[index].bulletSize;
296     /**
297      * Dano de la bala por defecto 0.
298      * @type Number
299      */
300     this.damage = (damage == null) ? 0 : damage;
301     /**
302      * Imagen de la bala coje la src del json tipotower.
303      * @type Image
304      */
305     this.img = new Image();
306     this.img.src = tipotower[index].imgbullet;
307     /**
308      * Enemigo que tiene seleccionado solo lo utilizara si el  bulletType es "target"
309      * @type Enemy
310      */
311     this.Enemy = Enemy;
312     /**
313      * Velocidad de la bala coje el valor del json tipotower.
314      * @type Number
315      */
316     this.velocity = tipotower[index].velocity;
317     this.radio = tipotower[index].radio;
318     this.countradio = 0;
319     
320     /**
321      * Metodo que comprueba si se ha chocado con otro cuerpo
322      * @param {type} rect
323      * @returns {boolean}
324      */
325     this.intersects = function(rect) {
326         if (rect != null) {
327             return(this.x <= rect.x + rect.width &&
328                     this.x + this.width > rect.x &&
329                     this.y <= rect.y + rect.height &&
330                     this.y + this.height > rect.y);
331         }
332     }
333 
334 /**
335  * Metodo para mover la bala. Actualiza su posicion.
336  * Si bulletType es target seguira al Enemy que tiene.
337  * Si bulletType es lineal o area la bala mirara que direccion tiene y se movera en esa direccion.
338  * @returns {void}
339  */
340     this.move = function() {
341         if (this.bulletType == "target") {
342             var dx = Enemy.x + Enemy.width / 2 - this.x;
343             var dy = Enemy.y + Enemy.height / 2 - this.y;
344             var angle = Math.atan(dy / dx);
345             var dxf = Math.cos(angle) * this.velocity;
346             var dyf = Math.sin(angle) * this.velocity;
347 
348             if (dx < 0) {
349                 dxf = dxf * -1;
350                 dyf = dyf * -1;
351             }
352 
353             this.x += dxf;
354             this.y += dyf;
355 
356         } else if (this.bulletType == "lineal" || this.bulletType == "area") {
357             switch (this.direccion) {
358                 case "top":
359                     this.y -= this.velocity;
360                     break;
361                 case "left":
362                     this.x -= this.velocity;
363                     break;
364                 case "bot":
365                     this.y += this.velocity;
366                     break;
367                 case "right":
368                     this.x += this.velocity;
369                     break;
370             }
371             this.countradio += this.velocity;
372 
373         }
374 
375 
376     }

464         if (this.Enemy != null) {
465             if (this.bulletType == "target") {
466                 ArrayBullet.push(new Bullet(this.x, this.y, this.Enemy, this.damage, index));
467 
468             } else if (this.bulletType == "lineal" || this.bulletType == "area") {
469                 ArrayBullet.push(new Bullet(this.x, this.y, this.Enemy, this.damage, index, "top"));
470                 ArrayBullet.push(new Bullet(this.x, this.y, this.Enemy, this.damage, index, "bot"));

472                 ArrayBullet.push(new Bullet(this.x, this.y, this.Enemy, this.damage, index, "right"));

474             }

481 /**
482  * Metodo que selecciona el target mas cercano dentro del radio para comenzar a disparar
483  * @returns {void}
484  */
485     this.selecttarget = function() {
486         if (this.Enemy == null && this.count >= this.delay) {
487             var modulomin = this.radio;
488 
489             for (var i = 0; i < horda.ArrayEnemy.length; i++) {
490                 var modulo = Math.sqrt(Math.pow(horda.ArrayEnemy[i].x - (this.x), 2) + Math.pow(horda.ArrayEnemy[i].y - (this.y), 2));
491                 if (modulo <= modulomin) {
492                     modulomin = modulo;
493                     this.Enemy = new Enemy();
494                     this.Enemy = horda.ArrayEnemy[i];
495                     this.count = 0;
496 
497                 }
498 
499             }
500             this.shoot();
501 
502         }
503 
504         this.count += 30;
505     }
